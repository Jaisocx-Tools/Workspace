let globTagsQuerySelector = ".wordlist-item";

function getTagsTextsAsArray( querySelector ) {
  let tags = document.querySelectorAll( querySelector );

  let locTagsTexts = new Array( tags.length );

  let tag = new Object();
  
  let tagId = 0;
  for ( tagId = 0; tagId < tags.length; tagsId++ ) {
    tag = tags[tagId];
    locTagsTexts[tagId] = tag.innerText;
  }

  return locTagsTexts;
}
